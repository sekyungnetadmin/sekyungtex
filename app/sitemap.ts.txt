export default function sitemap() {
  const baseUrl = "https://sekyungnet.vercel.app"; // 배포 후 실제 URL로 변경

  return [
    {
      url: `${baseUrl}/`,
      lastModified: new Date(),
      changeFrequency: "monthly",
      priority: 1.0,
    },
    { url: `${baseUrl}/ko`, lastModified: new Date() },

    // Static pages
    { url: `${baseUrl}/ko/company`, lastModified: new Date() },
    { url: `${baseUrl}/ko/facilities`, lastModified: new Date() },
    { url: `${baseUrl}/ko/contact`, lastModified: new Date() },

    // Product pages
    { url: `${baseUrl}/ko/products/golf`, lastModified: new Date() },
    { url: `${baseUrl}/ko/products/sports`, lastModified: new Date() },
    { url: `${baseUrl}/ko/products/ski`, lastModified: new Date() },
    { url: `${baseUrl}/ko/products/safety`, lastModified: new Date() },
    { url: `${baseUrl}/ko/products/fishery`, lastModified: new Date() },
  ];
}
